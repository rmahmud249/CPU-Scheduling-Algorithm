#include <stdio.h>
#include <pthread.h>

int a = 48, b = 18;
int result;

void* find_gcd(void* arg) {
    int x = a, y = b;

    while (y != 0) {
        int temp = y;
        y = x % y;
        x = temp;
    }

    result = x;
    return NULL;
}

int main() {
    pthread_t t;

    pthread_create(&t, NULL, find_gcd, NULL);
    pthread_join(t, NULL);

    printf("GCD = %d\n", result);

    return 0;
}
